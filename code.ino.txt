String formatTime(unsigned long totalSecs) { 
  int h = totalSecs / 3600; 
  int m = (totalSecs % 3600) / 60; 
  int s = totalSecs % 60; 
  char buf[10]; 
  sprintf(buf, "%02d:%02d:%02d", h, m, s); 
  return String(buf); 
} 
 
void setLoadCurrent(float amps) { 
  if (amps < 0.01) { 
    ledcWrite(PWM_PIN, 0);  
    targetCurrentAmps = 0; 
    return; 
  } 
  float requiredPwmPercent = 0.0; 
  if (amps <= 0.6) { 
    requiredPwmPercent = 11.5 * (amps * amps) + 31.5 * amps + 0.2; 
  } else { 
    requiredPwmPercent = 23.2 + ((amps - 0.6) * highCurrentSlope);  
  } 
  requiredPwmPercent = constrain(requiredPwmPercent, 0.0, 100.0); 
  int mappedDuty = map(requiredPwmPercent, 0, 100, 0, 255); 
  ledcWrite(PWM_PIN, mappedDuty); 
  targetCurrentAmps = amps; 
} 
 
void setCharger(bool on) { digitalWrite(RELAY_PIN, on ? HIGH : LOW); } 
bool isChargeFull() { return digitalRead(LED_BLUE_PIN) == LOW; } 
 
void beep(int times, int durationMs) { 
  for (int i = 0; i < times; i++) { 
    digitalWrite(BUZZER_PIN, HIGH); delay(durationMs); 
digitalWrite(BUZZER_PIN, LOW); 
    if (i < times - 1) delay(durationMs); 
  } 
} 
 
float getVoltage() { 
  int16_t adc0 = ads.readADC_SingleEnded(0); 
  return ads.computeVolts(adc0);  
} 
 
int getBatteryPercentage(float volts) { 
  int pct = map((long)(volts * 100), 300, 420, 0, 100); 
  return constrain(pct, 0, 100); 
}

void handleParams() { 
  if (server.hasArg("cur")) { 
    int newCur = server.arg("cur").toInt(); 
    if(newCur >= 100 && newCur <= 2000) targetCurrent_mA = newCur; 
  } 
  if (server.hasArg("cut")) { 
    float newCut = server.arg("cut").toFloat(); 
    if(newCut >= 2.5 && newCut <= 4.2) targetCutoff_V = newCut; 
  } 
  saveSettingsToFlash(); 
   
  if (currentState == DSG_QUICK || currentState == NOR_DSG) { 
    setLoadCurrent(targetCurrent_mA / 1000.0); 
  } 
  server.send(200, "text/plain", "OK"); 
} 
 
void handleControl() { 
  String cmd = server.arg("cmd"); 
   
  if (cmd == "fan") { 
    manualFanOverride = !manualFanOverride; 
    beep(1, 100); 
  } 
  else if (cmd == "stop") { 
    beep(1, 600); 
    setLoadCurrent(0); setCharger(false); 
    lastSavedCap = capacity_mAh; lastSavedSeconds = totalTestSeconds; 
    prefs.putFloat("lastCap", lastSavedCap); prefs.putUInt("lastSecs", 
lastSavedSeconds); 
    saveSettingsToFlash();  
    statusMessage = "Stopped.";  
    currentState = FINISHED; 
  } 
  else if (cmd == "start" && (currentState == IDLE_MENU || 
currentState == FINISHED || currentState == SHOW_LAST_RESULT)) { 
    int mode = server.arg("mode").toInt(); 
    beep(2, 100);  
    capacity_mAh = 0; totalTestSeconds = 0; histCount = 0;  
    lastGraphSave = millis(); lastNvsSave = millis(); stateStartMillis 
= millis();  
    chargeFullDetectMillis = 0; liveAdjustCutoff = false;  
     
    if (mode == 0) { prefs.putFloat("lastCap", 0.0); 
prefs.putUInt("lastSecs", 0); setLoadCurrent(targetCurrent_mA / 
1000.0); currentState = DSG_QUICK; }  
    else if (mode == 1) { prefs.putFloat("lastCap", 0.0); 
prefs.putUInt("lastSecs", 0); currentState = NOR_CHG1; statusMessage = 
"Charging..."; } 
    else if (mode == 2) { prefs.putUInt("lastSecs", 0); currentState = 
CHG_ONLY; statusMessage = "Charging..."; } 
    else if (mode == 3) {  
      prefs.putUInt("lastSecs", 0); 
      if (batteryVoltage > 3.9) { setLoadCurrent(targetCurrent_mA / 
1000.0); currentState = STORE_DSG; } 
      else if (batteryVoltage < 3.8) { currentState = STORE_CHG; 
statusMessage = "Store: Charging..."; } 
      else { currentState = FINISHED; statusMessage = "Stored!"; 
beep(3, 100); } 
    } 
    else if (mode == 4) { currentState = MEASURE_IR; } 
  } 
  server.send(200, "text/plain", "OK"); 
}

void handleButtons() { 
  if (millis() - lastButtonPress < 150) return;  
 
  if (currentState == FINISHED || currentState == SHOW_LAST_RESULT) { 
    if (digitalRead(BTN_BACK) == LOW || digitalRead(BTN_OK) == LOW) { 
      beep(1, 50); lastButtonPress = millis(); 
      currentState = IDLE_MENU; statusMessage = "Ready."; return; 
    } 
  } 
 
  if (currentState == IDLE_MENU) { 
    if (digitalRead(BTN_DOWN) == LOW) { 
      beep(1, 30); lastButtonPress = millis(); 
      if (!inSubMenu) { menuCursor++; if(menuCursor >= 
TOTAL_MENU_ITEMS) menuCursor = 0; } 
      else {  
        if (menuCursor == 6) { targetCurrent_mA -= 50; 
if(targetCurrent_mA < 100) targetCurrent_mA = 100; } 
        if (menuCursor == 7) { targetCutoff_V -= 0.1; 
if(targetCutoff_V < 2.5) targetCutoff_V = 2.5; } 
        if (menuCursor == 8) { irPulseAmps -= 0.1; if(irPulseAmps < 
0.1) irPulseAmps = 0.1; } 
      } 
    } 
    if (digitalRead(BTN_UP) == LOW) { 
      beep(1, 30); lastButtonPress = millis(); 
      if (!inSubMenu) { menuCursor--; if(menuCursor < 0) menuCursor = 
TOTAL_MENU_ITEMS - 1; } 
      else {  
        if (menuCursor == 6) { targetCurrent_mA += 50; 
if(targetCurrent_mA > 1500) targetCurrent_mA = 1500; } 
        if (menuCursor == 7) { targetCutoff_V += 0.1; 
if(targetCutoff_V > 3.5) targetCutoff_V = 3.5; } 
        if (menuCursor == 8) { irPulseAmps += 0.1; if(irPulseAmps > 
2.0) irPulseAmps = 2.0; } 
      } 
    } 
    if (digitalRead(BTN_OK) == LOW) { 
      beep(1, 30); lastButtonPress = millis(); 
      if (!inSubMenu && (menuCursor == 6 || menuCursor == 7 || 
menuCursor == 8)) { inSubMenu = true; }  
      else if (inSubMenu) {  
        inSubMenu = false; saveSettingsToFlash(); beep(2, 40);  
      }  
    } 
    if (digitalRead(BTN_BACK) == LOW && inSubMenu) {  
      beep(1,30); inSubMenu = false; lastButtonPress = millis();  
    } 
  } 
   
  bool canAdjustLive = (currentState == DSG_QUICK || currentState == 
NOR_DSG || currentState == NOR_CHG1 || currentState == NOR_REST1); 
   
  if (canAdjustLive) { 
    if (digitalRead(BTN_OK) == LOW) { 
      beep(1, 40); lastButtonPress = millis(); 
      liveAdjustCutoff = !liveAdjustCutoff;  
    } 
    if (digitalRead(BTN_UP) == LOW) { 
      beep(1, 30); lastButtonPress = millis(); 
      if (liveAdjustCutoff) {  
        targetCutoff_V += 0.1; if(targetCutoff_V > 4.2) targetCutoff_V 
= 4.2;  
      } else {  
        targetCurrent_mA += 50; if(targetCurrent_mA > 2000) 
targetCurrent_mA = 2000;  
        if (currentState == DSG_QUICK || currentState == NOR_DSG) { 
setLoadCurrent(targetCurrent_mA / 1000.0); } 
      } 
    } 
    if (digitalRead(BTN_DOWN) == LOW) { 
      beep(1, 30); lastButtonPress = millis(); 
      if (liveAdjustCutoff) {  
        targetCutoff_V -= 0.1; if(targetCutoff_V < 2.5) targetCutoff_V 
= 2.5;  
      } else {  
        targetCurrent_mA -= 50; if(targetCurrent_mA < 100) 
targetCurrent_mA = 100;  
        if (currentState == DSG_QUICK || currentState == NOR_DSG) { 
setLoadCurrent(targetCurrent_mA / 1000.0); } 
      } 
    } 
  } 
 
  if (digitalRead(BTN_STARTSTOP) == LOW) { 
    lastButtonPress = millis(); 
     
    if (currentState == IDLE_MENU && !inSubMenu) { 
      beep(2, 100);  
      capacity_mAh = 0; totalTestSeconds = 0; histCount = 0;  
      lastGraphSave = millis(); lastNvsSave = millis(); 
stateStartMillis = millis();  
      chargeFullDetectMillis = 0; liveAdjustCutoff = false;  
       
      if (menuCursor == 0) { prefs.putFloat("lastCap", 0.0); 
prefs.putUInt("lastSecs", 0); setLoadCurrent(targetCurrent_mA / 
1000.0); currentState = DSG_QUICK; }  
      else if (menuCursor == 1) { prefs.putFloat("lastCap", 0.0); 
prefs.putUInt("lastSecs", 0); currentState = NOR_CHG1; statusMessage = 
"Charging..."; } 
      else if (menuCursor == 2) { prefs.putUInt("lastSecs", 0); 
currentState = CHG_ONLY; statusMessage = "Charging..."; } 
      else if (menuCursor == 3) {  
        prefs.putUInt("lastSecs", 0); 
        if (batteryVoltage > 3.9) { setLoadCurrent(targetCurrent_mA / 
1000.0); currentState = STORE_DSG; } 
        else if (batteryVoltage < 3.8) { currentState = STORE_CHG; 
statusMessage = "Store: Charging..."; } 
        else { currentState = FINISHED; statusMessage = "Stored!"; 
beep(3, 100); } 
      } 
      else if (menuCursor == 4) { currentState = MEASURE_IR; } 
      else if (menuCursor == 5) { currentState = SHOW_LAST_RESULT; 
statusMessage = "Viewing Last Result"; } 
    }  
    else if (currentState > IDLE_MENU && currentState != FINISHED && 
currentState != SHOW_LAST_RESULT) { 
      beep(1, 600); 
      setLoadCurrent(0); setCharger(false); 
      lastSavedCap = capacity_mAh; lastSavedSeconds = 
totalTestSeconds; 
      prefs.putFloat("lastCap", lastSavedCap); 
prefs.putUInt("lastSecs", lastSavedSeconds); 
      saveSettingsToFlash();  
      statusMessage = "Stopped.";  
      currentState = FINISHED; 
    } 
  } 
} 
 
// ========================================== 
// SYSTEM SETUP 
// ========================================== 
void setup() { 
  Serial.begin(115200); 
   
  pinMode(BUZZER_PIN, OUTPUT); digitalWrite(BUZZER_PIN, LOW); 
  pinMode(PWM_PIN, OUTPUT);    digitalWrite(PWM_PIN, LOW); 
  pinMode(RELAY_PIN, OUTPUT);  digitalWrite(RELAY_PIN, LOW);  
  pinMode(FAN_PIN, OUTPUT);    digitalWrite(FAN_PIN, LOW); 
 
  pinMode(BTN_UP, INPUT_PULLUP); pinMode(BTN_DOWN, INPUT_PULLUP); 
  pinMode(BTN_OK, INPUT_PULLUP); pinMode(BTN_BACK, INPUT_PULLUP); 
  pinMode(BTN_STARTSTOP, INPUT_PULLUP); pinMode(LED_BLUE_PIN, 
INPUT_PULLUP);  
 
  u8g2.begin(); 
  Wire.setClock(400000);  
  if (!ads.begin()) { Serial.println("ADS1115 fail"); } 
  ads.setGain(GAIN_TWOTHIRDS);  
   
  prefs.begin("batTest", false); 
  targetCurrent_mA = prefs.getInt("setCur", 500); 
  targetCutoff_V = prefs.getFloat("setCut", 3.0); 
  irPulseAmps = prefs.getFloat("setIrCur", 0.5); 
   
  lastSavedCap = prefs.getFloat("lastCap", 0.0); 
  lastSavedIR = prefs.getFloat("lastIR", 0.0); 
  lastSavedSeconds = prefs.getUInt("lastSecs", 0); 
   
  ledcAttach(PWM_PIN, PWM_FREQ, PWM_RES); 
  setLoadCurrent(0);  
 
  // --- START WIFI & WEB SERVER --- 
  WiFi.softAP(AP_SSID, AP_PASS); 
  server.on("/", handleRoot); 
  server.on("/api/status", handleStatus); 
  server.on("/api/graph", handleGraph); 
  server.on("/api/params", handleParams); 
  server.on("/api/control", handleControl); 
  server.begin(); 
 
  beep(3, 80); 
  statusMessage = "Insert Battery"; 
} 
 
// ========================================== 
// MAIN CONTROL LOOP 
// ========================================== 
void loop() { 
  server.handleClient();  
  handleButtons(); 
  batteryVoltage = getVoltage(); 
  unsigned long currentMillis = millis(); 
 
  bool isTestActive = (currentState >= DSG_QUICK && currentState <= 
STORE_DSG); 
 
  if (manualFanOverride || currentState == DSG_QUICK || currentState 
== NOR_DSG || currentState == STORE_DSG) { 
    digitalWrite(FAN_PIN, HIGH);  
  } else { 
    digitalWrite(FAN_PIN, LOW);   
  } 
 
  // --- CAPACITY, TIMER, AND GRAPH TRACKING --- 
  if (isTestActive) { 
    if (currentMillis - previousMillis >= 1000) { 
      previousMillis = currentMillis; 
      totalTestSeconds++; 
      if (currentState == DSG_QUICK || currentState == NOR_DSG || 
currentState == STORE_DSG) { 
        capacity_mAh += targetCurrentAmps * (1000.0 / 3600.0); 
      } 
    } 
     
    if (currentMillis - lastGraphSave >= 60000) { 
      lastGraphSave = currentMillis; 
      if (histCount < MAX_POINTS) { 
        vHist[histCount] = batteryVoltage; 
        iHist[histCount] = targetCurrentAmps;  
        histCount++; 
      } 
    } 
     
    if (currentMillis - lastNvsSave >= 60000) { 
      lastNvsSave = currentMillis; 
      lastSavedCap = capacity_mAh; 
      lastSavedSeconds = totalTestSeconds; 
      prefs.putFloat("lastCap", lastSavedCap); 
      prefs.putUInt("lastSecs", lastSavedSeconds); 
    } 
  } 
 
  // --- STATE MACHINE --- 
  switch (currentState) { 
    case WAIT_BATTERY: 
      if (batteryVoltage > 2.5) { delay(500); currentState = 
MEASURE_IR; } 
      break; 
 
    case MEASURE_IR:  
      statusMessage = "Testing IR..."; beep(1, 150);  
      setLoadCurrent(irPulseAmps); delay(100);  
      { 
        float rawIR = ((batteryVoltage - getVoltage()) / irPulseAmps) 
* 1000.0; 
        internalResistance = rawIR - wiringResistance_mR; 
        if(internalResistance < 0) internalResistance = 0;  
        lastSavedIR = internalResistance; prefs.putFloat("lastIR", 
lastSavedIR); 
      } 
      setLoadCurrent(0); statusMessage = "Ready."; currentState = 
IDLE_MENU; 
      break; 
 
    case IDLE_MENU: break; 
    case SHOW_LAST_RESULT: break; 
 
    case DSG_QUICK: 
    case NOR_DSG:  
      statusMessage = "Discharging..."; 
      if (batteryVoltage <= targetCutoff_V) {  
        setLoadCurrent(0);  
        lastSavedCap = capacity_mAh; lastSavedSeconds = 
totalTestSeconds; 
        prefs.putFloat("lastCap", lastSavedCap); 
prefs.putUInt("lastSecs", lastSavedSeconds); saveSettingsToFlash();  
         
        if (currentState == DSG_QUICK) { statusMessage = "Complete."; 
currentState = FINISHED; beep(3, 300); }  
        else { statusMessage = "NOR: Rest 2"; stateStartMillis = 
currentMillis; currentState = NOR_REST2; } 
      } 
      break; 
 
    case NOR_CHG1: 
    case NOR_CHG2: 
    case CHG_ONLY: 
      setCharger(true); 
      if (currentMillis - stateStartMillis > 3000) { 
        if (isChargeFull()) {  
          if (chargeFullDetectMillis == 0) { chargeFullDetectMillis = 
currentMillis; }  
          else if (currentMillis - chargeFullDetectMillis >= 10000) { 
            setCharger(false); chargeFullDetectMillis = 0; 
            if (currentState == NOR_CHG1) { statusMessage = "NOR: Rest 
1"; stateStartMillis = currentMillis; capacity_mAh = 0; currentState = 
NOR_REST1; } 
            else { statusMessage = "Charge Complete!"; currentState = 
FINISHED; beep(4, 200); } 
          } else { statusMessage = "Charge Full (Wait 10s)"; } 
        } else { chargeFullDetectMillis = 0; statusMessage = 
"Charging..."; } 
      } 
      break; 
 
    case NOR_REST1: 
    case NOR_REST2: 
      statusMessage = "Cooldown (5s)..."; 
      if (currentMillis - stateStartMillis >= 5000) { 
        stateStartMillis = currentMillis;  
        if(currentState == NOR_REST1) {  
          prefs.putFloat("lastCap", 0.0); 
setLoadCurrent(targetCurrent_mA / 1000.0); currentState = NOR_DSG;  
        } else { currentState = NOR_CHG2; statusMessage = 
"Charging..."; } 
      } 
      break; 
 
    case STORE_CHG: 
      setCharger(true); 
      if (currentMillis - stateStartMillis > 3000) { 
        if (batteryVoltage >= 3.85) { setCharger(false); statusMessage 
= "Storage Set!"; currentState = FINISHED; beep(3, 200); }  
        else if (isChargeFull()) { 
          if (chargeFullDetectMillis == 0) { chargeFullDetectMillis = 
currentMillis; }  
          else if (currentMillis - chargeFullDetectMillis >= 10000) { 
            setCharger(false); chargeFullDetectMillis = 0; 
statusMessage = "Storage Set!"; currentState = FINISHED; beep(3, 200); 
          } else { statusMessage = "Charge Full (Wait 10s)"; } 
        } else { chargeFullDetectMillis = 0; statusMessage = "Store: 
Charging..."; } 
      } 
      break; 
 
    case STORE_DSG: 
      statusMessage = "Store: Discharging..."; 
      if (batteryVoltage <= 3.85) { setLoadCurrent(0); statusMessage = 
"Storage Set!"; currentState = FINISHED; beep(3, 200); } 
      break; 
 
    case FINISHED:  
      if (batteryVoltage < 1.0) { currentState = WAIT_BATTERY; 
statusMessage = "Insert Battery"; } 
      break; 
  } 
 
  // --- OLED DRAWING LOOP (30 FPS) --- 
  if (millis() - lastOledUpdate >= 33) { 
    lastOledUpdate = millis(); 
    u8g2.clearBuffer(); 
     
    // ========================================== 
    // THE GRAPH & RESULTS SCREEN 
    // ========================================== 
    if (currentState == FINISHED) { 
      u8g2.setFont(u8g2_font_ncenB08_tr); 
       
      u8g2.setCursor(0, 10); u8g2.print("Cap:"); 
u8g2.print((int)capacity_mAh); 
      String timeStr = formatTime(totalTestSeconds); 
      u8g2.setCursor(128 - u8g2.getStrWidth(timeStr.c_str()), 10); 
u8g2.print(timeStr); 
       
      u8g2.setCursor(0, 21); u8g2.print(statusMessage); 
      String irStr = String((int)internalResistance) + "mR"; 
      u8g2.setCursor(128 - u8g2.getStrWidth(irStr.c_str()), 21); 
u8g2.print(irStr); 
       
      if (histCount > 1) { 
        float minV = 5.0, maxV = 0.0; 
        for(int i=0; i<histCount; i++) { 
          if(vHist[i] < minV) minV = vHist[i]; 
          if(vHist[i] > maxV) maxV = vHist[i]; 
        } 
         
        u8g2.drawLine(16, 26, 16, 63);  
        u8g2.drawLine(16, 63, 127, 63);  
         
        u8g2.setFont(u8g2_font_micro_tr);  
        u8g2.setCursor(0, 31); u8g2.print(maxV, 1); 
        u8g2.setCursor(0, 63); u8g2.print(minV, 1); 
         
        for (int i = 0; i < histCount - 1; i++) { 
          int x1 = map(i, 0, histCount - 1, 18, 127);  
          int x2 = map(i + 1, 0, histCount - 1, 18, 127); 
          int y1 = map((long)(vHist[i]*1000), (long)(maxV*1000), 
(long)(minV*1000), 26, 62);  
          int y2 = map((long)(vHist[i+1]*1000), (long)(maxV*1000), 
(long)(minV*1000), 26, 62); 
          u8g2.drawLine(x1, y1, x2, y2); 
        } 
      } else { u8g2.setCursor(0, 50); u8g2.print("No graph data."); } 
    }  
    // ========================================== 
    // "LAST RESULT" RECOVERY SCREEN 
    // ========================================== 
    else if (currentState == SHOW_LAST_RESULT) { 
      u8g2.setFont(u8g2_font_ncenB10_tr); 
      String header = "== LAST RESULT =="; 
      int headX = (128 - u8g2.getStrWidth(header.c_str())) / 2; 
      u8g2.setCursor(headX, 15); u8g2.print(header); 
       
      u8g2.setFont(u8g2_font_ncenB12_tr); 
      u8g2.setCursor(5, 36); u8g2.print("Cap: " + 
String((int)lastSavedCap) + " mAh"); 
      u8g2.setCursor(5, 50); u8g2.print("IR  : " + 
String((int)lastSavedIR) + " mR"); 
      u8g2.setFont(u8g2_font_ncenB10_tr); 
      u8g2.setCursor(5, 63); u8g2.print("Time: " + 
formatTime(lastSavedSeconds)); 
    } 
    else { 
      // ========================================== 
      // STANDARD UI SCREENS 
      // ========================================== 
      int rawPct = getBatteryPercentage(batteryVoltage); 
      if (smoothPercentage < 0) { smoothPercentage = rawPct; }  
      else { smoothPercentage += (rawPct - smoothPercentage) * 0.05; } 
 
      u8g2.setFont(u8g2_font_ncenB08_tr); 
      u8g2.setCursor(0, 14); u8g2.print((int)smoothPercentage); 
u8g2.print("%"); 
       
      String irStr = String((int)internalResistance) + "mR"; 
      u8g2.setCursor(128 - u8g2.getStrWidth(irStr.c_str()), 14); 
u8g2.print(irStr); 
       
      u8g2.setFont(u8g2_font_ncenB14_tr); 
      String volStr = String(batteryVoltage, 2) + "V"; 
      u8g2.setCursor((128 - u8g2.getStrWidth(volStr.c_str())) / 2, 
16); u8g2.print(volStr); 
       
      u8g2.drawLine(0, 19, 128, 19); 
 
      if (currentState == WAIT_BATTERY) { 
        u8g2.setFont(u8g2_font_ncenB10_tr); u8g2.setCursor(10, 45); 
u8g2.print("Insert Battery"); 
      }  
      else if (currentState == IDLE_MENU) { 
        if (!inSubMenu) { 
          float targetScroll = menuCursor * 15.0; animatedScroll += 
(targetScroll - animatedScroll) * 0.4;  
          u8g2.setFont(u8g2_font_ncenB08_tr); 
          for(int i = 0; i < TOTAL_MENU_ITEMS; i++) { 
            float yPos = 36 + (i * 15) - animatedScroll; 
            if(yPos > 28 && yPos < 75) { 
              if(i == menuCursor) { u8g2.drawStr(2, (int)yPos, ">"); } 
              u8g2.drawStr(12, (int)yPos, menuItems[i]); 
            } 
          } 
        }  
        else { 
          u8g2.setFont(u8g2_font_ncenB10_tr); 
          if (menuCursor == 6) { u8g2.setCursor(0, 36); 
u8g2.print("Set: "); u8g2.print(targetCurrent_mA); u8g2.print(" mA"); 
}  
          else if (menuCursor == 7) { u8g2.setCursor(0, 36); 
u8g2.print("Cutoff: "); u8g2.print(targetCutoff_V, 1); u8g2.print(" 
V"); }  
          else if (menuCursor == 8) { u8g2.setCursor(0, 36); 
u8g2.print("IR Test: "); u8g2.print(irPulseAmps, 1); u8g2.print(" A"); 
} 
          u8g2.setFont(u8g2_font_ncenB08_tr); u8g2.setCursor(10, 58); 
u8g2.print("[UP/DWN] to Save [OK]"); 
        } 
      }  
      else { 
        bool canAdjustLive = (currentState == DSG_QUICK || 
currentState == NOR_DSG || currentState == NOR_CHG1 || currentState == 
NOR_REST1); 
        u8g2.setFont(u8g2_font_ncenB08_tr); 
        u8g2.setCursor(0, 32); u8g2.print("Cap: "); 
u8g2.print((int)capacity_mAh);  
         
        String liveTime = formatTime(totalTestSeconds); 
        u8g2.setCursor(128 - u8g2.getStrWidth(liveTime.c_str()), 32); 
u8g2.print(liveTime); 
         
        u8g2.setCursor(0, 46);  
        if(canAdjustLive && !liveAdjustCutoff) { u8g2.print(">"); } 
        u8g2.print(targetCurrent_mA / 1000.0, 2); u8g2.print("A  ");  
         
        u8g2.setCursor(65, 46);  
        if(canAdjustLive && liveAdjustCutoff) { u8g2.print(">"); } 
        u8g2.print(targetCutoff_V, 1); u8g2.print("V"); 
         
        u8g2.drawLine(0, 52, 128, 52); u8g2.setCursor(0, 63); 
u8g2.print(statusMessage); 
      } 
    } 
    u8g2.sendBuffer(); 
  } 
}
